f1[X_] := X^7
f2[X_] := P[3, X, 50]
f3[X_] := f1[X] - f2[X]
errorPercent[X_] := Abs[(f3[X]/f1[X])*100]

(*Generate the table with step 0.1 from 0.5 to 4*)
step = 0.1;
range = Range[20, 90, step];
table = Table[{x, f1[x], f2[x], f3[x], errorPercent[x]}, {x, range}];

(*Print the table*)
Grid[Prepend[table, {"X", "X^7", "P", "Diff", "ABS Error %"}],
 Frame -> All]
